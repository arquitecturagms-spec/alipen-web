async function procesarImagen() { const fileInput = document.getElementById('input-foto'); const btn = document.getElementById('btn-generar'); const loading = document.getElementById('loading'); const imgDisplay = document.getElementById('resultado-img'); if (fileInput.files.length === 0) return alert("Selecciona una foto."); btn.disabled = true; loading.style.display = "block"; imgDisplay.style.display = "none"; const formData = new FormData(); formData.append('image', fileInput.files[0]); try { const response = await fetch("https://alipen-arte-mi-disenador-ia.hf.space/remodelar", { method: "POST", body: formData, mode: 'cors', // Forzamos el modo CORS headers: { "Accept": "image/*, application/json" } }); if (!response.ok) { const errorText = await response.text(); throw new Error(errorText || "Error en el servidor"); } const blob = await response.blob(); const imageUrl = URL.createObjectURL(blob); imgDisplay.src = imageUrl; imgDisplay.style.display = "block"; } catch (error) { console.error("Detalle del error:", error); alert("Error de comunicación. Prueba lo siguiente:\n1. Pulsa Ctrl+F5\n2. Asegúrate de que la foto no sea demasiado pesada."); } finally { btn.disabled = false; loading.style.display = "none"; } }
